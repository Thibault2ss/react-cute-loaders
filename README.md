## Install
with npm:
```bash
npm install react-cute-loaders
```

with yarn:
```bash
yarn add react-cute-loaders
```

## Usage

```tsx
import React from 'react'
import {  Cube3DLoader  } from 'react-cute-loaders'

const YourComponent = () => {
   return <Cube3DLoader size={40} variant={1} color='#e20b78' />
}
```
